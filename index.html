    const tbody = document.getElementById("tbody");

    const search = document.getElementById("search");

    const clearBtn = document.getElementById("clearBtn");

    const countPill = document.getElementById("countPill");

    const headers = Array.from(document.querySelectorAll("thead .th-btn"));


    let sortKey = "name";

    let sortDir = "asc";

    let query = "";


    function render() {

      const q = query.trim().toLowerCase();

      let rows = DATA.filter(d =>

        d.name.toLowerCase().includes(q) ||

        d.url.toLowerCase().includes(q) ||

        d.types.toLowerCase().includes(q)

      );


      rows.sort((a,b) => {

        const A = (a[sortKey] || "").toLowerCase();

        const B = (b[sortKey] || "").toLowerCase();

        if (A < B) return sortDir === "asc" ? -1 : 1;

        if (A > B) return sortDir === "asc" ? 1 : -1;

        return 0;

      });


      tbody.innerHTML = rows.map(d => `

        <tr>

          <td><strong>${escapeHtml(d.name)}</strong></td>

          <td><a class="link" href="${escapeAttr(d.url)}" target="_blank" rel="noopener noreferrer">${escapeHtml(d.url)}</a></td>

          <td>${escapeHtml(d.types)}</td>

        </tr>

      `).join("");


      countPill.textContent = `${rows.length} item${rows.length!==1?'s':''}`;

    }


    function escapeHtml(str){

      return String(str).replace(/[&<>'\"]/g, s => ({'&':'&amp;','<':'&lt;','>':'&gt;',\"'\":'&#39;','\"':'&quot;'}[s]));

    }

    function escapeAttr(str){ return escapeHtml(str); }


    function setSort(key){

      if (sortKey === key) {

        sortDir = sortDir === "asc" ? "desc" : "asc";

      } else {

        sortKey = key;

        sortDir = "asc";

      }

      headers.forEach(h=>{

        h.classList.toggle("asc", h.dataset.sort===sortKey && sortDir==="asc");

        h.classList.toggle("desc", h.dataset.sort===sortKey && sortDir==="desc");

      });

      render();

    }


    headers.forEach(h => h.addEventListener("click", () => setSort(h.dataset.sort)));

    search.addEventListener("input", e => { query = e.target.value; render(); });

    clearBtn.addEventListener("click", () => { search.value = ""; query = ""; render(); });


    // initial header state + render

    headers.forEach(h=>{

      h.classList.toggle("asc", h.dataset.sort===sortKey && sortDir==="asc");

      h.classList.toggle("desc", h.dataset.sort===sortKey && sortDir==="desc");

    });

    render();
